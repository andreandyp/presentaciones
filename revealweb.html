    console.log("Hola");

    console.log("¿Cómo estas?");
else
    console.log("El triple igual es diferente");

//Clases ECMAScript 6
class Coche{
	constructor(modelo,año){
		this.modelo = modelo;
		this.año = año;
	}
	avanzar(){
		console.log("Estoy avanzando");
	}
}
var coche = new Coche("Tesla Modelo S",2014);
coche.avanzar();
console.log(coche.modelo);

//Parametros por valor y por referencia
function Coche() {
	this.año = arguments[0];
	this.marca = arguments[1];
	this.modelo = arguments[2];
}
function ejecutar() {
	var variable = 23;
	var carrito = new Coche(1996,"Dodge","Viper R/T");
	console.log("Valor de hue antes de la función: "+variable);
	console.log("Valor de Coche.año antes de la función: "+carrito.año);
	modificar(variable,carrito);
	console.log("Valor de hue después de la función: "+variable);
	console.log("Valor de Coche.año después de la función: "+carrito.año);
}
function modificar(hue,objeto) {
	hue = 24;
	objeto.año = 1997;
}
ejecutar();

Complementos: RevealEditor.js

Abrir editor

Incluye atajos de teclado:

  • f: pantalla completa (Full-screen)
  • s: modo de presentador (Speaker mode)
  • o: Ver todas las diapositivas (Overview)
  • b: pausar presentación (Black screen)

Editando el código fuente Usando el editor oficial

En este caso, demostraré la primera opción 😆

Sin NPM 😢

(Pero con Github 😁)

En 4 pasos:

  1. Descargar desde Github
    • Repositorio de Reveal.js (Releases)
    • https://github.com/hakimel/reveal.js/releases
    • Source Code (zip)
  2. Descomprimir
  3. Usa tu editor favorito
    • Visual Studio Code
    • Sublime Text
    • Brackets
  4. ¡A programar!

Con NPM 🤗

En 5 pasos:

  1. Instala Git
  2. Instala Node.js
  3. Ejecuta en este orden (Git)
    • git clone https://github.com/hakimel/reveal.js.git
    • cd reveal.js
  4. Ejecuta en este orden (NPM)
    • npm install
    • npm start
  5. ¡A programar!
